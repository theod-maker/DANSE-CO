    **src/lib/pageBuilderQueries.ts** :
    ```ts
    import { groq } from '@sanity/client'

    export const pageQuery = groq`*[_type == "page" && slug == $slug][0]{
      _id, title, slug,
      blocks[] {
        _type, _key,
        // heroBlock
        tagline, title, description,
        "imageUrl": image.asset->url,
        ctaLabel, ctaLink,
        // pageHeaderBlock
        eyebrow, titleHighlight, subtitle,
        // richTextBlock
        content,
        // imageBlock
        "imageUrl": image.asset->url,
        caption, fullWidth,
        // galleryBlock
        "imageUrls": images[].asset->url,
        columns,
        // ctaBlock
        buttonLabel, buttonLink,
        // autres champs partagés
      }
    }`
    ```

    Note: GROQ projections dans un array polymorphe — les champs sont projetés sur tous les types, les champs manquants retournent null. C'est le comportement normal de Sanity.

    **src/hooks/usePage.ts** :
    ```ts
    import { useState, useEffect } from 'react'
    import { sanityClient, isSanityConfigured } from '../lib/sanityClient'
    import { pageQuery } from '../lib/pageBuilderQueries'
    import type { PageContent } from '../lib/pageBuilderTypes'

    export function usePage(slug: string): PageContent | null {
      const [page, setPage] = useState<PageContent | null>(null)
      useEffect(() => {
        if (!isSanityConfigured) return
        sanityClient.fetch<PageContent>(pageQuery, { slug })
          .then(result => { if (result) setPage(result) })
          .catch(() => {})
      }, [slug])
      return page
    }
    ```

    import type { PageBlock } from '../../lib/pageBuilderTypes'
    // imports de tous les blocs simples + lazy imports pour les blocs complexes

    interface Props { blocks: PageBlock[] }

    const BlockRenderer: React.FC<Props> = ({ blocks }) => (
      <>
        {blocks.map(block => {
          switch (block._type) {
            case 'heroBlock': return <HeroBlock key={block._key} {...block} />
            case 'pageHeaderBlock': return <PageHeaderBlock key={block._key} {...block} />
            case 'richTextBlock': return <RichTextBlock key={block._key} {...block} />
            case 'imageBlock': return <ImageBlock key={block._key} {...block} />
            case 'galleryBlock': return <GalleryBlock key={block._key} {...block} />
            case 'ctaBlock': return <CtaBlock key={block._key} {...block} />
            case 'dividerBlock': return <DividerBlock key={block._key} />
            // blocs complexes — placeholders pour le plan 06-03
            case 'disciplinesGridBlock':
            case 'instructorsGridBlock':
            case 'scheduleBlock':
            case 'pricingBlock':
            case 'venuesBlock':
            case 'newsBlock':
            case 'contactBlock':
              return <div key={block._key} data-block-type={block._type} />
            default: return null
          }
        })}
      </>
    )
    export default BlockRenderer
