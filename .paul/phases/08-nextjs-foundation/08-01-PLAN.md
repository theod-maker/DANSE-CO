  <name>Task 2: Créer next.config.ts</name>
  <files>next.config.ts</files>
  <action>
    Créer next.config.ts à la racine :

    ```ts
    import type { NextConfig } from 'next'

    const config: NextConfig = {
      transpilePackages: ['sanity'],
      images: {
        remotePatterns: [
          { protocol: 'https', hostname: 'cdn.sanity.io' },
          { protocol: 'https', hostname: 'images.unsplash.com' },
        ],
      },
    }

    export default config
    ```

    Sanity nécessite transpilePackages pour éviter les erreurs ESM.
    Les deux hostnames couvrent les images Sanity et les fallbacks Unsplash existants.
  </action>
  <verify>cat next.config.ts</verify>
  <done>Configuration Next.js présente avec transpile Sanity</done>
</task>
